## SQL LIKE

특정 패턴과 일치하는 값을 검색하기 위해 사용되는 연산자이다.

``` sql
/* users 테이블에서 email이 a로 시작하는 행의 데이터만 가져오기 */
SELECT * FROM users WHERE email LIKE "a%";

/* users 테이블에서 email이 gmail.com으로 끝나는 행의 데이터만 가져오기 */
SELECT * FROM users WHERE email LIKE "%gmail.com";

/* users 테이블에서 email에 co를 포함하는 행의 데이터만 가져오기 */
SELECT * FROM users WHERE email LIKE "%co%";

/* users 테이블에서 email의 두 번째 문자가 r인 행의 데이터만 가져오기 */
SELECT * FROM users WHERE email LIKE "_r%";

/* users 테이블에서 email이 a로 시작하면서 길이가 2 이상인 행의 데이터만 가져오기 */
SELECT * FROM users WHERE email LIKE "a_%";
```
